def []= (i,j, num) 
        insertado = false
        if(num == 0)
                for a in(0...@fil.size)
                        if (@vfil[a] == i && @vcol[a] == j)
                                @vfil.delete_at(a)
                                @vcol.delete_at(a)
                                @vval.delete_at(a)
                                insertado = true
                        end
                end
        else
                for a in(0...@fil.size)
                        if (@vfil[a] == i && @vcol[a] == j) 
                                @vval[a]=num
                                insertado = true
                        end
                end
        end
        if(insertado == false)
                @vval.push num
                @vfil.push i
                @vcol.push j
        end
end
